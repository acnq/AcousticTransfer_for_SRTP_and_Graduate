$$
\begin{aligned}
 & S_0^0(\bold{x_i }- \bold{x_{0j}}) &&= \frac{1}{2}\sqrt{\frac{1}{\pi}}\frac{i}{kr}e^{-ikr} \\

& S_{1}^{0} &&= -\frac{1}{2}\sqrt{\frac{3}{2\pi}}\frac{z}{r}\frac{kr - i}{(kr)^2}e^{-ikr} \\

 \end{aligned}
